// CASO 1: Correcto - Declaración simple de variable

// CASO 2: Error - falta punto y coma

// CASO 3: Correcto - Declaración con operación

// CASO 4: Error - paréntesis sin cerrar en estructura si

    x = x + 1;
}

// CASO 5: Correcto - Función con 2 parámetros

    entero resultado = a + b;
}

// CASO 6: Correcto - Estructura si-sino
si (x > 10) {
    x = 0;
} sino {
    x = x + 1;
}

// CASO 7: Error - Falta llave de cierre

    x = x + 1;

// CASO 8: Correcto - Estructura mientras
mientras (x < 50) {
    x = x + 5;
}

// CASO 9: Correcto - Estructura hacer-mientras
hacer {
    x = x - 1;
} mientras (x > 0)

// CASO 10: Error - Tipo de dato inválido

// CASO 12: Correcto - Expresión compleja con paréntesis
entero complejo = (10 + 5) * 2 - 3;

// CASO 13: Error - Comparador inválido (tres iguales)
si (x === 10) {
    x = 0;
}

// CASO 14: Correcto - Función con 3 parámetros
decimal promedio(decimal a, decimal b, decimal c) {
    decimal suma = a + b + c;
    decimal prom = suma / 3.0;
}
